python 
import requests

API = "https://domains.purpleflea.com/v1"
HEADERS = {"Authorization": "Bearer sk_live_..."}

for name in trending_names:
    r = requests.post(f"{API}/domains/search",
        json={"query": name, "tlds": [".ai"]},
        headers=HEADERS)
    if r.json()["results"][0]["available"]:
